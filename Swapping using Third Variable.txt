//Swapping of two numbers using third Variable
public class Swappings

{
    public static void main(String[]args){
        int a = 10;
        int b = 20;
        int c;

        c=a;
        a=b;
        b=c;
        System.out.println("Value of a after swapping is " +a);
        System.out.println("Value of b after swapping is " +b);
    }
}
